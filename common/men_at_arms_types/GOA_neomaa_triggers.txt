# Basic Triggers

can_recruit_elite_maa = {
  OR = {
    highest_held_title_tier >= tier_kingdom
    AND = {
      is_independent_ruler = yes
      highest_held_title_tier >= tier_duchy
    }
  }
  ATHA_maa_elite_count < ATHA_maa_elite_maximum
}

can_recruit_guard_maa = {
  is_independent_ruler = yes
  highest_held_title_tier >= tier_kingdom
  ATHA_maa_guard_count < ATHA_maa_guard_maximum
}

can_recruit_culture_heritage_maa_trigger = {
  OR = {
    culture = { has_cultural_pillar = $HERITAGE$ }
    any_directly_owned_province = {
      culture = { has_cultural_pillar = $HERITAGE$ }
    }
  }
}

can_recruit_culture_maa_trigger = {
  OR = {
    has_culture_or_parent_culture = { CULTURE = $CULTURE$ }
    any_directly_owned_province = {
        has_culture_or_parent_culture = { CULTURE = $CULTURE$ }
    }
  }
}

can_recruit_culture_aor_maa_trigger = {
  OR = {
    AND = {
      has_culture_or_parent_culture = { CULTURE = $CULTURE$ }
      any_directly_owned_province = {
        geographical_region = $REGION$
      }
    }
    any_directly_owned_province = {
      has_culture_or_parent_culture = { CULTURE = $CULTURE$ }
      geographical_region = $REGION$
    }
  }
}

### Heritages

can_recruit_amphibian_maa_trigger = {
  can_recruit_culture_heritage_maa_trigger = { HERITAGE = heritage_amphibian }
}

can_recruit_orcish_maa_trigger = {
  can_recruit_culture_heritage_maa_trigger = { HERITAGE = heritage_orcish }
}

can_recruit_human_maa_trigger = {
  OR = {
    can_recruit_culture_heritage_maa_trigger = { HERITAGE = heritage_northern }
    can_recruit_culture_heritage_maa_trigger = { HERITAGE = heritage_arathi }
    can_recruit_culture_heritage_maa_trigger = { HERITAGE = heritage_gilnean }
    can_recruit_culture_heritage_maa_trigger = { HERITAGE = heritage_southern }
  }
}
