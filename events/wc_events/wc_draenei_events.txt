namespace = WCDRN

############################
#	Crashlanding of the Exodar
############################

# Creates the Draenei
WCDRN.1 = {
	type = empty
	hidden = yes
	
	trigger = {
		NOT = { exists = global_var:exodar_has_crashed }
		year = 610
	}

	immediate = {
		set_global_variable =  {
			name = exodar_has_crashed
			value = yes
		}
		debug_log = "The Exodar has crashlanded on Azeroth!"
		debug_log_date = yes
		spawn_draenei_effect = yes
	}
}

# Notify the world of the crash, give initial boost for Velen
WCDRN.2 = {
	title = WCDRN.2.title
	desc = WCDRN.2.desc
	theme = diplomacy
	override_background = {
		reference = wc_background_exodar_crashsite
	}
	right_portrait = {
		character = scope:velen
		animation = disbelief
	}

	option = {
		name = WCDRN.2.a
		trigger = {
			OR = {
				has_religion = religion:light_group
				has_religion = religion:naarism
			}
		}
	}
	
	option = {
		name = WCDRN.2.b
		trigger = { 
			OR = { 
				faith = faith:burning_legion_religion
				faith = faith:orcish_fel
			} 
		}		
	}

	option = {
		name = WCDRN.2.c
		trigger = { has_culture_group = culture_group:orc_group }
	}

	option = {	
		name = WCDRN.2.d
		trigger = { 
			NOR = { 
				has_religion = religion:light_group
				has_religion = religion:naarism			
				has_culture_group = culture_group:orc_group
				faith = faith:burning_legion_religion
				faith = faith:orcish_fel
			} 
		}
	}
}
